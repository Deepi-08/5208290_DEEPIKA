Time Conversion

#include<stdio.h>
#include<string.h>
int main(){
    char time[11];
    int hour;
    scanf("%s",&time);
    sscanf(time, "%2d",&hour);
    if (time[8] == 'A') {
        if (hour == 12) hour = 0;
    } else {
        if (hour != 12) hour += 12;
    }
    printf("%02d:%c%c:%c%c\n", hour, time[3], time[4], time[6], time[7]);
}
